    c.id id,
    c.size size,
    c.content content,
    c.binary binary,
    c.copies copies,
FROM [ornate-woodland-196220:github_demo.files] f

    SELECT * 
    FROM [bigquery-public-data:github_repos.contents] 
  ) c
ON f.id = c.id
